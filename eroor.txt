export class YourComponent implements OnInit {
  page: number = 1;         // Current page for first table
  page1: number = 1;        // Current page for second table
  noOfRows: number = 5;     // Rows per page for first table
  noOfRows1: number = 5;    // Rows per page for second table

  // Pagination config
  config: PaginationInstance = {
    id: 'pagination1',      // ID for the first table pagination
    itemsPerPage: this.noOfRows,
    currentPage: this.page
  };

  config1: PaginationInstance = {
    id: 'pagination2',      // ID for the second table pagination
    itemsPerPage: this.noOfRows1,
    currentPage: this.page1
  };
  
  // Pagination logic methods
  createRange(lastPage: number): any {
    let paginationArray: any = [];
    for (let i = 0; i < lastPage; i++) {
      paginationArray.push({
        label: `${i + 1}`,
        value: i + 1
      });
    }
    return paginationArray;
  }

  ngOnInit() {
    // Initialize other component logic as needed
  }
}
