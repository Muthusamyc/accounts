sendemailfrom(response: any, status: any) {
  // Retrieve data from response
  const crrequestedBy = this.Cremailvalue[0].crrequestedBy;
  const crdate = this.Cremailvalue[0].crdate;
  const changeDesc = this.Cremailvalue[0].changeDesc;

  // Read HTML file
  const output = this.readHtmlFile('assets/email.html');

  // Replace template variables with actual data
  const populatedOutput = output
    .replace('{{this.Cremailvalue[0].crrequestedBy}}', crrequestedBy)
    .replace('{{this.Cremailvalue[0].crdate}}', crdate)
    .replace('{{this.Cremailvalue[0].changeDesc}}', changeDesc)
    .replace('${status}', status); // Replace ${status} as well

  // Construct email body
  const to = this.Cremailvalue[0].crremail;
  const cc = this.Cremailvalue[0].croemail;
  const subject = `Unnati: IT Change Request:${this.Cremailvalue[0].itcrid} : Submitted for Approval`;

  const requestBody = {
    to: to,
    cc: cc,
    subject: subject,
    body: populatedOutput // Use the populated HTML
  };

  // Send the email
  const httpOptions = {
    headers: new HttpHeaders({
      'Content-Type': 'application/json'
    })
  };

  this.http.post(apiUrl, requestBody, httpOptions).subscribe(
    (response: any) => {
      console.log(response);
    },
    (error: any) => {
      console.log('Post request failed', error);
    }
  );
}
