public async Task<IActionResult> UploadFile(int itcrid, IFormFile file)
{
    try
    {
        if (file == null || file.Length == 0)
        {
            return BadRequest("File is not selected or empty");
        }

        if (itcrid <= 0)
        {
            return BadRequest("ITCRID is required and must be a positive integer");
        }

        var uploadsFolder = Path.Combine(Directory.GetCurrentDirectory(), "Uploads", itcrid.ToString());
        if (!Directory.Exists(uploadsFolder))
        {
            Directory.CreateDirectory(uploadsFolder);
        }

        var fileName = file.FileName; // Use the original file name provided by the user
        var filePath = Path.Combine(uploadsFolder, fileName);

        using (var stream = new FileStream(filePath, FileMode.Create))
        {
            await file.CopyToAsync(stream);
        }

        return Ok(new { fileName });
    }
    catch (Exception ex)
    {
        return StatusCode(StatusCodes.Status500InternalServerError, $"Internal server error: {ex.Message}");
    }
}
