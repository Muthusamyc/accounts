mapedplantdatas() {
  this.impactedLocation = this.selectedPlantIds.map((item: any) => item.item_id);
  console.log(this.impactedLocation);
  this.selectedlocationNames = Array.from(new Set(this.impactedLocation));
}

filterdonutbyplant() {
  if (this.selectedlocationNames.length > 0) { // Check if there are selected locations
    // Filter the data based on selected location names
    this.filterdata = this.changerequest.filter((item: any) => this.selectedlocationNames.includes(item.plantcode));
    this.changerequest = this.filterdata;
  } else {
    // If no locations are selected, reset the filter to show all data
    this.filterdata = this.changerequest;
  }
  console.log("here is multi donut selected plantid", this.changerequest);
}
