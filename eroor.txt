filterdonutbyplant() {
  if (this.selectedlocationNames.length > 0) {
    // Filter the selectedPlantIds based on the filtered code values
    this.selectedPlantIds = this.selectedlocationNames.filter(item =>
      this.code.some(codeItem => codeItem.plantId === item)
    );
  } else {
    // If no filters applied, reset selectedPlantIds
    this.selectedPlantIds = [];
  }
  console.log("Selected PlantIds:", this.selectedPlantIds);
}
