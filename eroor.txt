ngOnInit(): void {
  this.page = 1;
  this.page1 = 1;
  this.noOfRows = 5;
  this.noOfRows1 = 5;
  this.config.itemsPerPage = this.noOfRows;
  this.config.currentPage = this.page;
  this.config1.itemsPerPage = this.noOfRows1;
  this.config1.currentPage = this.page1;
}





ngOnChanges(changes: SimpleChanges): void {
  if (changes.noOfRows) {
    this.config.itemsPerPage = this.noOfRows;
    this.page = 1; // Reset to the first page when items per page changes
  }
  if (changes.noOfRows1) {
    this.config1.itemsPerPage = this.noOfRows1;
    this.page1 = 1; // Reset to the first page when items per page changes
  }
}
