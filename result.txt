import { HttpClient } from '@angular/common/http';

export class YourComponent {
  data: any = [];
  items: any[] = [];
  selectedCategory: string = ''; 

  constructor(private http: HttpClient) {
    this.getData();
  }

  getData() {
    const apiUrl = 'https://localhost:7236/api/SystemLandscape';
    this.http.get(apiUrl).subscribe(
      (response: any) => {
        console.log(response);
        this.data = response;
        this.filterItems(); // Filter items after fetching data
      },
      (error: any) => {
        console.error('GET request failed', error);
      }
    );
  }

  filterItems() {
    // Filter the items based on the selected category
    this.items = this.data.filter(item => item.sysLandscape1.trim() === this.selectedCategory.trim());
  }
}
