 GetApprover() {
   const apiUrl = this.apiurl + '/GetApproval/GetApprover';
   const requestBody = {
     "level": 1,
     "stage": "N",
     "crid": this.itcrid,
   }
   const httpOptions = {
     headers: new HttpHeaders({
       'Content-Type': 'application/json'
     })
   };

   this.http.post(apiUrl, requestBody, httpOptions).subscribe(
     (response: any) => {

       this.approvervalue = response;

       if (this.approvervalue.approver1 == '') {
         this.approvervalue[0][1] = this.approvervalue.approver1Name
       }

       console.log("Approver", this.approvervalue, this.approverCount)
     },
     (error: any) => {
       console.error('POST request failed', error);
     });
 }
