import { Component } from '@angular/core';

@Component({
  selector: 'app-your-component',
  templateUrl: './your-component.component.html',
  styleUrls: ['./your-component.component.css']
})
export class YourComponent {
  selectedCategory: string = 'development'; // Default selected category
  data: any[] = []; // Your original data array
  filteredData: any[] = []; // Array to hold filtered results

  constructor() {
    // Assign your original data array to the filteredData array initially
    this.filteredData = this.data.filter(item => item.sysLandscape1.trim().toLowerCase() === this.selectedCategory.trim().toLowerCase());
  }

  // Function to filter data based on selected category
  filterData() {
    this.filteredData = this.data.filter(item => item.sysLandscape1.trim().toLowerCase() === this.selectedCategory.trim().toLowerCase());
  }
}


<Select class="" name="category" [(ngModel)]="selectedCategory" (ngModelChange)="filterData()">
  <option>Select Category</option>
  <option value="Development">Development</option>
  <option value="Testing">Testing</option>
  <option value="Validation">Validation</option>
  <option value="Production">Production</option>
</Select>
