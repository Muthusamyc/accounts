<Select class="" name="category" [(ngModel)]="selectedCategory">
  <option>Development</option>
  <option>Quality</option>
  <option>Validation</option>
  <option>Production</option>
</Select>

<!-- Assuming your items are stored in a variable called 'items' -->
<div *ngFor="let item of filteredItems">
  {{ item.sysLandscape1 }}
</div>


// In your component file
export class YourComponent {
  selectedCategory: string; // Variable to store the selected category
  items: any[] = [...]; // Your array of items

  get filteredItems() {
    // Filter the items based on the selected category
    return this.items.filter(item => item.sysLandscape1.trim() === this.selectedCategory.trim());
  }
}
